MINILIBX_PATH = ./minilibx-linux
MINILIBX = $(MINILIBX_PATH)/libmlx.a
LIBFT_PATH = ./libft
LIBFT = $(LIBFT_PATH)/libft.a

SOURCES_FILES	=	main.c\
					check_map.c\
					get_next_line.c\
					init_map.c\
					gameplay.c\
					movement.c\
					image_builder.c\
					enemy_animations.c\
					exit_code.c\
					moves_print.c\

HEADER = ./so_long.h

OBJECTS = $(SOURCES_FILES:.c=.o)

NAME = so_long

CC = gcc

RM = rm -f

CFLAGS = -Wall -Wextra -Werror

%.o: %.c
	$(CC) $(CFLAGS) -Imlx_linux -c $< -o $@

all: $(NAME)

$(NAME): $(OBJECTS) $(LIBFT) $(MINILIBX)
	$(CC) $(OBJECTS) $(LIBFT) $(MINILIBX) -L -lmlx -lXext -lX11 -lm -o $(NAME)

$(MINILIBX):
	make -C $(MINILIBX_PATH)

$(LIBFT):
	make -C $(LIBFT_PATH)

clean:
	$(RM) $(OBJECTS)
	make clean -C $(MINILIBX_PATH)
	make clean -C $(LIBFT_PATH)

fclean: clean
	$(RM) $(NAME)
	$(RM) $(MINILIBX)
	$(RM) $(LIBFT)

re: fclean all

.PHONY: all clean fclean re
